### 动态尺寸计算

```typescript
import {Dimensions} from 'react-native';

const {width: SCREEN_WIDTH, height: SCREEN_HEIGHT} = Dimensions.get('window');

// 海报尺寸
const POSTER_WIDTH = SCREEN_WIDTH - 32; // 左右各16px边距
const POSTER_HEIGHT = SCREEN_HEIGHT * 0.75; // 屏幕高度的75%
const POSTER_ASPECT_RATIO = POSTER_WIDTH / POSTER_HEIGHT;

// 更新 posterTheme
export const posterDimensions = {
  width: POSTER_WIDTH,
  height: POSTER_HEIGHT,
  aspectRatio: POSTER_ASPECT_RATIO,
} as const;
```

### 标签分布网格组件
```typescript
// 网格布局：每行显示固定数量的方块
const GRID_COLUMNS = 12; // 每行12个方块
const GRID_ROWS = 10;    // 共10行

// 根据标签数据生成网格
const renderTagGrid = () => {
  const totalCells = GRID_COLUMNS * GRID_ROWS;
  const cells = [];
  
  // 根据标签占比分配方块
  let currentIndex = 0;
  tagDistribution.forEach(tag => {
    const cellCount = Math.round((tag.percentage / 100) * totalCells);
    for (let i = 0; i < cellCount; i++) {
      cells.push({
        color: tag.color,
        tagName: tag.tag_name,
      });
    }
  });
  
  // 渲染网格
  return (
    <View style={styles.grid}>
      {cells.map((cell, index) => (
        <View
          key={index}
          style={[
            styles.gridCell,
            {backgroundColor: cell.color},
          ]}
        />
      ))}
    </View>
  );
};
```
