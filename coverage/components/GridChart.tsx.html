/**
 * 生成渐变色系 - 使用 gluestack-ui 颜色风格
 * @param count 需要生成的颜色数量
 * @param isRed 是否是红色系（true=红色，false=青色）
 *
 * 注意: 使用 RGB 函数生成渐变色，这是必要的设计需求
 * 起止颜色值来自 gluestack-ui 的颜色系统：
 * - 起始色: #09090B (surface, backgroundDark900)
 * - 红色终点: #EF4444 (error500)
 * - 青色终点: #00D9FF (primary500)
 *
 * gluestack-ui 设计规范：
 * - 红色系：从深色背景到 error500 (#EF4444)
 * - 青色系：从深色背景到 primary500 (#00D9FF)
 * - 颜色梯度模拟 GitHub 贡献图，采用 gluestack-ui 配色
 */
const generateGradientColors = (count: number, isRed: boolean): string[] => {
  const colors: string[] = [];
 
  if (isRed) {
    // 红色系（加班）：从深黑到亮红
    const startR = 9,
      startG = 9,
      startB = 11; // #09090B (surface)
    const endR = 239,
      endG = 68,
      endB = 68; // #EF4444 (destructive)
 
    for (let i = 0; i < count; i++) {
      const ratio = count === 1 ? 1 : i / (count - 1);
      const r = Math.round(startR + (endR - startR) * ratio);
      const g = Math.round(startG + (endG - startG) * ratio);
      const b = Math.round(startB + (endB - startB) * ratio);
      colors.push(`rgb(${r}, ${g}, ${b})`);
    }
  } else {
    // 青色系（下班）：从深黑到亮青
    const startR = 9,
      startG = 9,
      startB = 11; // #09090B (surface)
    const endR = 0,
      endG = 217,
      endB = 255; // #00D9FF (primary)
 
    for (let i = 0; i < count; i++) {
      const ratio = count === 1 ? 1 : i / (count - 1);
      const r = Math.round(startR + (endR - startR) * ratio);
      const g = Math.round(startG + (endG - startG) * ratio);
      const b = Math.round(startB + (endB - startB) * ratio);
      colors.push(`rgb(${r}, ${g}, ${b})`);
    }
  }
 
  return colors;
};
 
/**
 * 处理Top10标签并合并其他类别
 * 验证需求: 9.4, 9.5
 */
const processTagDistribution = (
  tagDistribution: TagDistribution[],
): GridItem[] => {
  console.log('[GridChart] Processing tag distribution:', tagDistribution);
 
  // 过滤掉count为0的标签
  const validTags = tagDistribution.filter(tag => tag.count > 0);
  console.log('[GridChart] Valid tags after filtering:', validTags);
 
  // 分别处理加班和准时下班的标签
  const overtimeTags = validTags.filter(tag => tag.isOvertime);
  const onTimeTags = validTags.filter(tag => !tag.isOvertime);
 
  console.log('[GridChart] Overtime tags:', overtimeTags);
  console.log('[GridChart] On-time tags:', onTimeTags);
 
  // 按数量排序并取Top10
  const sortedOvertimeTags = [...overtimeTags].sort(
    (a, b) => b.count - a.count,
  );
  const sortedOnTimeTags = [...onTimeTags].sort((a, b) => b.count - a.count);
 
  const top10Overtime = sortedOvertimeTags.slice(0, 10);
  const top10OnTime = sortedOnTimeTags.slice(0, 10);
 
  // 生成渐变色
  const overtimeColors = generateGradientColors(top10Overtime.length, true);
  const onTimeColors = generateGradientColors(top10OnTime.length, false);
 
  const result: GridItem[] = [];
 
  // 先添加Top10准时下班标签（使用渐变绿色）
  top10OnTime.forEach((tag, index) => {
    result.push({
      id: tag.tagId,
      tagName: tag.tagName,
      color: onTimeColors[index],
      isOvertime: false,
      count: tag.count,
    });
  });
 
  // 再添加Top10加班标签（使用渐变红色）
  top10Overtime.forEach((tag, index) => {
    result.push({
      id: tag.tagId,
      tagName: tag.tagName,
      color: overtimeColors[index],
      isOvertime: true,
      count: tag.count,
    });
  });
 
  console.log('[GridChart] Processed result:', result);
  return result;
};
 
/**
 * 计算每个标签应该占用的方格数量
 * 使用改进的分配算法，确保所有方格都被分配
 */
const calculateGridAllocation = (
  items: GridItem[],
  totalCount: number,
  totalGrids: number,
): Map<string, number> => {
  const allocation = new Map<string, number>();
 
  if (totalCount === 0 || items.length === 0) {
    return allocation;
  }
 
  // 第一步：计算每个标签的精确比例和初始分配（向下取整）
  const allocations: Array<{
    id: string;
    exact: number;
    allocated: number;
    remainder: number;
  }> = items.map(item => {
    const exact = (item.count / totalCount) * totalGrids;
    const allocated = Math.floor(exact);
    return {
      id: item.id,
      exact,
      allocated: Math.max(1, allocated), // 至少分配1个
      remainder: exact - Math.floor(exact),
    };
  });
 
  // 第二步：计算已分配的总数
  let totalAllocated = allocations.reduce((sum, a) => sum + a.allocated, 0);
 
  // 第三步：如果总数不够，按余数从大到小补充
  if (totalAllocated < totalGrids) {
    const remaining = totalGrids - totalAllocated;
    // 按余数排序
    allocations.sort((a, b) => b.remainder - a.remainder);
 
    // 给余数最大的项补充方格
    for (let i = 0; i < remaining && i < allocations.length; i++) {
      allocations[i].allocated++;
    }
  }
  // 第四步：如果总数超了，从分配最多的项中减少
  else Iif (totalAllocated > totalGrids) {
    const excess = totalAllocated - totalGrids;
    // 按已分配数量从大到小排序
    allocations.sort((a, b) => b.allocated - a.allocated);
 
    // 从分配最多的项中减少（但至少保留1个）
    for (let i = 0; i < excess && i < allocations.length; i++) {
      if (allocations[i].allocated > 1) {
        allocations[i].allocated--;
      }
    }
  }
 
  // 第五步：构建最终的分配映射
  allocations.forEach(a => {
    allocation.set(a.id, a.allocated);
  });
 
  return allocation;
};
 
/**
 * GridSquare - 单个网格方块（gluestack-ui 风格）
 *
 * gluestack-ui 设计规范：
 * - 使用 gluestack-ui 的 borderRadius tokens
 * - 快速线性动画（100ms）
 * - 选中时使用 gluestack-ui 的 primary 颜色边框
 */
const GridSquare: React.FC<{
  color: string;
  delay: number;
  animationDuration: number;
  isSelected: boolean;
  isDimmed: boolean;
}> = ({color, delay, animationDuration, isSelected, isDimmed}) => {
  const opacity = useSharedValue(0);
 
  useEffect(() => {
    opacity.value = withTiming(1, {
      duration: 100, // 快速动画 100ms
      easing: Easing.linear, // 线性动画
    });
  }, [color]);
 
  const animatedStyle = useAnimatedStyle(() => {
    return {
      opacity: opacity.value,
    };
  });
 
  return (
    <Animated.View
      style={[
        styles.gridSquare,
        {backgroundColor: color},
        animatedStyle,
        isDimmed && styles.gridSquareDimmed,
        isSelected && styles.gridSquareSelected,
      ]}
    />
  );
};
 
export const GridChart = forwardRef<GridChartRef, GridChartProps>(
  (
    {
      tagDistribution,
      overtimeCount,
      onTimeCount,
      theme = 'light',
      animationDuration = 800,
    },
    ref,
  ) => {
    const screenWidth = Dimensions.get('window').width;
    const parentPadding = 16; // TrendPage scrollContent 的 padding
    const availableWidth = screenWidth - parentPadding * 2;
 
    // 选中的标签 ID
    const [selectedTagId, setSelectedTagId] = useState<string | null>(null);
 
    // 暴露清除选中的方法给父组件
    useImperativeHandle(ref, () => ({
      clearSelection: () => {
        setSelectedTagId(null);
      },
    }));
 
    // 计算可以容纳的列数
    const columns = Math.floor(
      (availableWidth + GRID_GAP) / (GRID_ITEM_SIZE + GRID_GAP),
    );
    const rows = 8; // 固定8行
    const totalGrids = columns * rows;
 
    // 处理标签分布
    const processedItems = useMemo(
      () => processTagDistribution(tagDistribution),
      [tagDistribution],
    );
 
    // 计算总数
    const totalCount = overtimeCount + onTimeCount;
 
    // 分配方格
    const gridAllocation = useMemo(
      () => calculateGridAllocation(processedItems, totalCount, totalGrids),
      [processedItems, totalCount, totalGrids],
    );
 
    // 生成网格数据（添加 tagId 字段用于识别标签）
    const gridData = useMemo(() => {
      const data: Array<{
        id: string;
        color: string;
        index: number;
        tagId: string;
      }> = [];
      let currentIndex = 0;
 
      processedItems.forEach(item => {
        const count = gridAllocation.get(item.id) || 0;
        for (let i = 0; i < count && currentIndex < totalGrids; i++) {
          data.push({
            id: `${item.id}-${i}`,
            color: item.color,
            index: currentIndex,
            tagId: item.id, // 添加标签 ID
          });
          currentIndex++;
        }
      });
 
      // 填充剩余空格，使用 surface 颜色（#09090B）
      const emptyColor = '#09090B'; // 硬核金融终端风格：统一使用 surface 颜色
      while (data.length < totalGrids) {
        data.push({
          id: `empty-${data.length}`,
          color: emptyColor,
          index: data.length,
          tagId: 'empty', // 空格子的标签 ID
        });
      }
 
      return data;
    }, [processedItems, gridAllocation, totalGrids, theme]);
 
    // 生成图例（显示所有标签，包含人数和占比）
    const legend = useMemo(() => {
      const legendItems = processedItems.map(item => {
        const percentage =
          totalCount > 0 ? ((item.count / totalCount) * 100).toFixed(1) : '0.0';
        return {
          id: item.id,
          name: item.tagName,
          color: item.color,
          count: item.count,
          percentage,
        };
      });
      console.log('[GridChart] Generated legend:', legendItems);
      return legendItems;
    }, [processedItems, totalCount]);
 
    // 处理格子点击
    const handleGridItemPress = (tagId: string) => {
      if (tagId === 'empty') {
        // 空格子点击 = 点击空白处，取消选中
        if (selectedTagId !== null) {
          setSelectedTagId(null);
        }
        return;
      }
      setSelectedTagId(selectedTagId === tagId ? null : tagId); // 切换选中状态
    };
 
    // 处理图例点击
    const handleLegendPress = (itemId: string) => {
      setSelectedTagId(selectedTagId === itemId ? null : itemId);
    };
 
    // 将网格数据按行分组
    const gridRows = useMemo(() => {
      const rows: Array<
        Array<{id: string; color: string; index: number; tagId: string}>
      > = [];
      for (let i = 0; i < gridData.length; i += columns) {
        rows.push(gridData.slice(i, i + columns));
      }
      return rows;
    }, [gridData, columns]);
 
    return (
      <Box w="$full">
        <VStack space="xs">
          {gridRows.map((row, rowIndex) => (
            <HStack key={`row-${rowIndex}`} space="xs">
              {row.map(item => {
                const isSelected = selectedTagId === item.tagId;
                const isDimmed = selectedTagId !== null && !isSelected;
 
                return (
                  <TouchableOpacity
                    key={item.id}
                    onPress={() => handleGridItemPress(item.tagId)}
                    activeOpacity={0.7}>
                    <GridSquare
                      color={item.color}
                      delay={item.index * 10}
                      animationDuration={animationDuration}
                      isSelected={isSelected}
                      isDimmed={isDimmed}
                    />
                  </TouchableOpacity>
                );
              })}
            </HStack>
          ))}
        </VStack>
 
        {legend.length > 0 && (
          <HStack flexWrap="wrap" space="xs" mt="$4">
            {legend.map(item => {
              const isSelected = selectedTagId === item.id;
              const isDimmed = selectedTagId !== null && !isSelected;
 
              return (
                <TouchableOpacity
                  key={item.id}
                  onPress={() => handleLegendPress(item.id)}
                  activeOpacity={0.7}>
                  <HStack
                    alignItems="center"
                    p="$1.5"
                    borderRadius="$sm"
                    bg={isSelected ? '$primary100' : '$backgroundLight50'}
                    opacity={isDimmed ? 0.4 : 1}>
                    <Box
                      w={14}
                      h={14}
                      borderRadius="$xs"
                      bg={item.color}
                      mr="$1.5"
                    />
                    <Text
                      size="xs"
                      color={theme === 'dark' ? '$textDark0' : '$textLight900'}
                      fontWeight={isSelected ? '$semibold' : '$normal'}>
                      {item.name} ({item.count}, {item.percentage}%)
                    </Text>
                  </HStack>
                </TouchableOpacity>
              );
            })}
          </HStack>
        )}
      </Box>
    );
  },
);
