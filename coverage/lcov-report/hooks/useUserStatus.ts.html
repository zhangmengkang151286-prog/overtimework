import {useEffect, useState, useCallback} from 'react';
import {useAppDispatch, useAppSelector} from './redux';
import {
  setUserSubmission,
  resetDailyStatus,
  setError,
} from '../store/slices/userSlice';
import {resetDailyData} from '../store/slices/dataSlice';
import {supabaseService} from '../services/supabaseService';
import {offlineQueueService} from '../services/offlineQueueService';
import {storageService} from '../services/storage';
import {UserStatusSubmission} from '../types';
import NetInfo from '@react-native-community/netinfo';
 
/**
 * 用户状态管理Hook
 * 处理用户状态提交、本地存储和每日重置逻辑
 * 验证需求: 7.4, 12.6
 */
export const useUserStatus = () => {
  const dispatch = useAppDispatch();
  const currentUser = useAppSelector(state => state.user.currentUser);
  const userStatus = useAppSelector(state => state.user.userStatus);
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [lastResetDate, setLastResetDate] = useState<string | null>(null);
 
  /**
   * 检查是否需要每日重置
   * 验证需求: 12.6
   */
  const checkDailyReset = useCallback(async () => {
    const today = new Date().toISOString().split('T')[0]; // YYYY-MM-DD
 
    // 从本地存储获取上次重置日期
    const storedResetDate = await storageService.getItem<string>(
      '@OvertimeIndexApp:lastResetDate',
    );
 
    if (storedResetDate !== today) {
      console.log('Performing daily reset...');
 
      // 执行每日重置
      dispatch(resetDailyStatus());
      dispatch(resetDailyData());
 
      // 保存新的重置日期
      await storageService.setItem('@OvertimeIndexApp:lastResetDate', today);
      setLastResetDate(today);
 
      // 清除本地存储的用户状态
      await storageService.saveUserStatus({
        hasSubmittedToday: false,
      });
    } else {
      setLastResetDate(storedResetDate);
    }
  }, [dispatch]);
 
  /**
   * 从本地存储恢复用户状态
   */
  const restoreUserStatus = useCallback(async () => {
    const storedStatus = await storageService.getUserStatus();
    if (storedStatus) {
      // 检查是否是今天的提交
      const today = new Date().toISOString().split('T')[0];
      const lastSubmissionDate = storedStatus.lastSubmission?.timestamp
        ? new Date(storedStatus.lastSubmission.timestamp)
            .toISOString()
            .split('T')[0]
        : null;
 
      if (lastSubmissionDate === today) {
        // 恢复今天的提交状态（Date 已经被序列化为字符串）
        dispatch(setUserSubmission(storedStatus.lastSubmission! as any));
      } else {
        // 如果不是今天的提交，重置状态
        dispatch(resetDailyStatus());
      }
    }
  }, [dispatch]);
 
  /**
   * 提交用户状态
   * 验证需求: 7.4, 14.6
   */
  const submitUserStatus = useCallback(
    async (submission: UserStatusSubmission) => {
      // 临时解决方案：如果没有登录用户，使用临时ID
      let userId = currentUser?.id;
      if (!userId) {
        // 生成或获取临时用户ID
        const tempUserId = await storageService.getItem<string>(
          '@OvertimeIndexApp:tempUserId',
        );
        if (tempUserId) {
          userId = tempUserId;
        } else {
          // 创建临时用户ID
          userId = `temp_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
          await storageService.setItem('@OvertimeIndexApp:tempUserId', userId);
          console.log('Created temporary user ID:', userId);
        }
      }
 
      setIsSubmitting(true);
 
      try {
        // 检查网络状态
        const netInfo = await NetInfo.fetch();
 
        if (netInfo.isConnected) {
          // 在线：直接提交到 Supabase
          await supabaseService.submitUserStatus({
            user_id: userId,
            date: submission.timestamp.toISOString().split('T')[0],
            is_overtime: submission.isOvertime,
            tag_id: submission.tagId,
            overtime_hours: submission.overtimeHours,
          });
        } else {
          // 离线：添加到队列
          await offlineQueueService.addToQueue('submitStatus', {
            userId: userId,
            isOvertime: submission.isOvertime,
            tagId: submission.tagId,
            overtimeHours: submission.overtimeHours,
            date: submission.timestamp.toISOString(),
          });
          console.log('Status added to offline queue');
        }
 
        // 序列化 submission（将 Date 转为字符串）
        const serializedSubmission = {
          ...submission,
          timestamp: submission.timestamp.toISOString(),
        };
 
        // 更新Redux状态（使用序列化后的数据）
        dispatch(setUserSubmission(serializedSubmission as any));
 
        // 保存到本地存储（使用序列化后的数据）
        await storageService.saveUserStatus({
          hasSubmittedToday: true,
          lastSubmission: serializedSubmission as any,
        });
 
        console.log('User status submitted successfully');
        return true;
      } catch (error) {
        console.error('Failed to submit user status:', error);
        dispatch(
          setError(
            error instanceof Error ? error.message : '提交状态失败，请重试',
          ),
        );
        return false;
      } finally {
        setIsSubmitting(false);
      }
    },
    [currentUser, dispatch],
  );
 
  /**
   * 初始化：检查每日重置和恢复用户状态
   */
  useEffect(() => {
    checkDailyReset();
    restoreUserStatus();
  }, [checkDailyReset, restoreUserStatus]);
 
  /**
   * 设置定时器在每日00:00执行重置
   * 验证需求: 12.1, 12.6
   */
  useEffect(() => {
    const checkResetInterval = setInterval(() => {
      checkDailyReset();
    }, 60 * 1000); // 每分钟检查一次
 
    return () => clearInterval(checkResetInterval);
  }, [checkDailyReset]);
 
  /**
   * 强制重置今日状态（用于调试）
   */
  const forceResetTodayStatus = useCallback(async () => {
    console.log('Force resetting today status...');
 
    // 重置 Redux 状态
    dispatch(resetDailyStatus());
 
    // 清除本地存储
    await storageService.saveUserStatus({
      hasSubmittedToday: false,
    });
 
    console.log('Today status has been reset');
  }, [dispatch]);
 
  return {
    userStatus,
    isSubmitting,
    submitUserStatus,
    forceResetTodayStatus,
    // 所有用户根据是否已提交决定是否显示提交按钮
    shouldShowSelector: !userStatus.hasSubmittedToday,
  };
};
