import {supabase, handleSupabaseError} from './supabase';
import {supabaseService} from './supabaseService';
import {storageService} from './storage';
import {User} from '../types';
 
/**
 * Supabase 认证服务
 * 处理用户注册、登录、会话管理
 */
class AuthService {
  // ============================================
  // 手机号认证
  // ============================================
 
  /**
   * 发送手机验证码
   * 注意：Supabase 默认不支持手机号验证码，需要配置第三方服务
   * 这里提供一个简化的实现方案
   */
  async sendPhoneVerificationCode(phoneNumber: string): Promise<void> {
    try {
      // 方案 1：使用 Supabase 的 OTP 功能（需要在 Dashboard 中启用）
      const {error} = await supabase.auth.signInWithOtp({
        phone: phoneNumber,
      });
 
      if (error) throw handleSupabaseError(error);
 
      console.log('验证码已发送到:', phoneNumber);
    } catch (error) {
      console.error('Send verification code error:', error);
      throw new Error('发送验证码失败，请稍后重试');
    }
  }
 
  /**
   * 验证手机号验证码并登录/注册
   */
  async verifyPhoneCode(
    phoneNumber: string,
    code: string,
  ): Promise<{user: User; isNewUser: boolean}> {
    try {
      // 使用 OTP 验证
      const {data, error} = await supabase.auth.verifyOtp({
        phone: phoneNumber,
        token: code,
        type: 'sms',
      });
 
      if (error) throw handleSupabaseError(error);
 
      if (!data.user) {
        throw new Error('验证失败');
      }
 
      // 检查用户是否已存在
      const existingUser = await supabaseService.getUserByPhone(phoneNumber);
 
      if (existingUser) {
        // 已存在的用户，直接登录
        await storageService.saveUser(existingUser);
        return {user: existingUser, isNewUser: false};
      } else {
        // 新用户，需要完善信息
        // 创建一个临时用户对象
        const tempUser: User = {
          id: data.user.id,
          phoneNumber: phoneNumber,
          avatar: '',
          username: '',
          province: '',
          city: '',
          industry: '',
          company: '',
          position: '',
          workStartTime: '09:00',
          workEndTime: '18:00',
          createdAt: new Date(),
          updatedAt: new Date(),
        };
 
        return {user: tempUser, isNewUser: true};
      }
    } catch (error) {
      console.error('Verify phone code error:', error);
      throw new Error('验证码错误或已过期');
    }
  }
 
  /**
   * 完善用户信息（新用户注册后）
   */
  async completeUserProfile(
    userId: string,
    profileData: {
      avatar?: string;
      username: string;
      province: string;
      city: string;
      industry: string;
      company: string;
      position: string;
      workStartTime: string;
      workEndTime: string;
    },
  ): Promise<User> {
    try {
      // 获取当前会话
      const {
        data: {session},
      } = await supabase.auth.getSession();
 
      if (!session) {
        throw new Error('用户未登录');
      }
 
      // 获取手机号
      const phoneNumber = session.user.phone || '';
 
      // 创建用户记录
      const user = await supabaseService.createUser({
        id: userId,
        phone_number: phoneNumber,
        avatar: profileData.avatar || '',
        username: profileData.username,
        province: profileData.province,
        city: profileData.city,
        industry: profileData.industry,
        company: profileData.company,
        position: profileData.position,
        work_start_time: profileData.workStartTime,
        work_end_time: profileData.workEndTime,
      });
 
      // 保存到本地存储
      await storageService.saveUser(user);
 
      return user;
    } catch (error) {
      console.error('Complete user profile error:', error);
      throw error;
    }
  }
 
  // ============================================
  // 微信登录（需要配置）
  // ============================================
 
  /**
   * 微信登录
   * 注意：需要在 Supabase Dashboard 中配置微信 OAuth
   */
  async signInWithWechat(): Promise<{user: User; isNewUser: boolean}> {
    try {
      // 使用 Supabase 的 OAuth 功能
      // 注意：这需要在 Supabase Dashboard 中配置微信作为 OAuth 提供商
      const {data, error} = await supabase.auth.signInWithOAuth({
        provider: 'wechat' as any, // Supabase 可能需要自定义配置
        options: {
          redirectTo: 'overtimeindex://auth/callback',
        },
      });
 
      if (error) throw handleSupabaseError(error);
 
      // 等待 OAuth 回调
      // 实际实现需要处理深度链接回调
      throw new Error('微信登录功能需要额外配置');
    } catch (error) {
      console.error('WeChat login error:', error);
      throw new Error('微信登录失败，请使用手机号登录');
    }
  }
 
  // ============================================
  // 会话管理
  // ============================================
 
  /**
   * 获取当前登录用户
   */
  async getCurrentUser(): Promise<User | null> {
    try {
      // 先从本地存储获取
      const localUser = await storageService.getUser();
      if (localUser) {
        return localUser;
      }
 
      // 检查 Supabase 会话
      const {
        data: {session},
      } = await supabase.auth.getSession();
 
      if (!session) {
        return null;
      }
 
      // 从数据库获取用户信息
      const user = await supabaseService.getUser(session.user.id);
 
      if (user) {
        await storageService.saveUser(user);
        return user;
      }
 
      return null;
    } catch (error) {
      console.error('Get current user error:', error);
      return null;
    }
  }
 
  /**
   * 检查用户是否已登录
   */
  async isAuthenticated(): Promise<boolean> {
    try {
      const {
        data: {session},
      } = await supabase.auth.getSession();
      return !!session;
    } catch (error) {
      console.error('Check authentication error:', error);
      return false;
    }
  }
 
  /**
   * 登出
   */
  async signOut(): Promise<void> {
    try {
      // 清除 Supabase 会话
      const {error} = await supabase.auth.signOut();
 
      if (error) throw handleSupabaseError(error);
 
      // 清除本地存储
      await storageService.clearUser();
 
      console.log('用户已登出');
    } catch (error) {
      console.error('Sign out error:', error);
      throw new Error('登出失败');
    }
  }
 
  /**
   * 刷新会话
   */
  async refreshSession(): Promise<void> {
    try {
      const {error} = await supabase.auth.refreshSession();
 
      if (error) throw handleSupabaseError(error);
 
      console.log('会话已刷新');
    } catch (error) {
      console.error('Refresh session error:', error);
      // 刷新失败不抛出错误，让用户继续使用
    }
  }
 
  /**
   * 监听认证状态变化
   */
  onAuthStateChange(
    callback: (event: string, session: any) => void,
  ): () => void {
    const {
      data: {subscription},
    } = supabase.auth.onAuthStateChange((event, session) => {
      console.log('Auth state changed:', event);
      callback(event, session);
    });
 
    // 返回取消订阅函数
    return () => {
      subscription.unsubscribe();
    };
  }
 
  // ============================================
  // 简化的认证方案（用于快速开发）
  // ============================================
 
  /**
   * 匿名登录（用于测试）
   * 创建一个匿名会话，用户可以使用应用但不保存个人信息
   */
  async signInAnonymously(): Promise<User> {
    try {
      const {data, error} = await supabase.auth.signInAnonymously();
 
      if (error) throw handleSupabaseError(error);
 
      if (!data.user) {
        throw new Error('匿名登录失败');
      }
 
      // 创建匿名用户对象
      const anonymousUser: User = {
        id: data.user.id,
        phoneNumber: '',
        avatar: '',
        username: '匿名用户',
        province: '未知',
        city: '未知',
        industry: '未知',
        company: '未知',
        position: '未知',
        workStartTime: '09:00',
        workEndTime: '18:00',
        createdAt: new Date(),
        updatedAt: new Date(),
      };
 
      await storageService.saveUser(anonymousUser);
 
      return anonymousUser;
    } catch (error) {
      console.error('Anonymous sign in error:', error);
      throw new Error('匿名登录失败');
    }
  }
 
  /**
   * 使用测试账号登录（用于开发）
   */
  async signInWithTestAccount(): Promise<User> {
    try {
      // 使用正确的 UUID 格式
      const testUserId = '00000000-0000-0000-0000-000000000001';
      
      // 创建本地测试用户（不依赖数据库查询，避免 RLS 问题）
      const testUser: User = {
        id: testUserId,
        phoneNumber: '13800138000',
        avatar: '',
        username: '测试用户',
        province: '北京市',
        city: '北京市',
        industry: '互联网',
        company: '测试公司',
        position: '软件工程师',
        workStartTime: '09:00',
        workEndTime: '18:00',
        createdAt: new Date(),

      };
 
      // 保存到本地存储
      await storageService.saveUser(testUser);

      return testUser;

      throw new Error('测试账号登录失败');
    }
  }
}
